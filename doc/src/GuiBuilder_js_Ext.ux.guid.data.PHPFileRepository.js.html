 19 Ext.namespace('Ext.ux.guid');
 20 Ext.namespace('Ext.ux.guid.data');
 21 
 22 /*
 23  * PHPFiles
 24  */
 25 Ext.ux.guid.data.PHPFileRepository = Ext.extend(Ext.ux.guid.data.Repository,{
 26   url : "phpFiles.php",  
 27   rootBase : "json",
 28   urlSupport : true,
 29     
 30   refresh : function (callback) {
 31     Ext.Ajax.request({
 32       url: this.url,
 33       params: {
 34          cmd: 'get_files',
 35          baseDir: this.rootBase
 36       },            
 37       callback: function(options, success, response) {
 38         this.items= success ? Ext.util.JSON.decode(response.responseText) : {};
 39         if(typeof callback == "function") callback(success);
 40       },            
 41       scope: this        
 42     });    
 43   },
 44 
 45   saveChanges : function(id,action,callback,content) {  
 46     Ext.Ajax.request({
 47        url: this.url,
 48        params: {
 49          cmd: 'save_changes',
 50          baseDir: this.rootBase,
 51          filename: id,
 52          action: action,
 53          content: content
 54        },
 55        callback: function(options, success, response) {
 56          if(success && response.responseText=='1') { 
 57            if(action=='remove') {
 58              delete this.items[id];
 59              if (id==this.last) this.last = null;
 60            } else {
 61              this.last = id;
 62            } 
 63          }
 64          if(typeof callback == "function") callback(response.responseText=='1');
 65        },
 66        scope: this        
 67     }); 
 68   },
 69 
 70   open : function(id,callback) {
 71     Ext.Ajax.request({
 72       url: this.url,
 73       params: {
 74         cmd: 'get_content',
 75         baseDir: this.rootBase,
 76         filename: id 
 77       },
 78       callback: function(options, success, response) {
 79         if (success) this.last = id;
 80         if(typeof callback == "function") callback(success,response.responseText);
 81       },
 82       scope: this        
 83     }); 
 84   }    
 85 });
