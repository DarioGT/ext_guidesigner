 11 Ext.extend(Ext.ux.tree.CodeLoader, Ext.util.Observable, {
 12  jsonId : '__JSON__',
 13  
 14  load : function(node, callback){ 
 15      while(node.firstChild) node.removeChild(node.firstChild);
 16      if(this.doLoad(node,this.designer.getConfig())){
 17        if(typeof callback == "function") callback();
 18      }
 19   },
 20   
 21   elementToText : function(c) {
 22       var txt = [];
 23       c = c || {};
 24       if (c.xtype)      { txt.push(c.xtype); }
 25       if (c.fieldLabel) { txt.push('[' + c.fieldLabel + ']'); }
 26       if (c.boxLabel)   { txt.push('[' + c.boxLabel + ']'); }
 27       if (c.layout)     { txt.push('<i>' + c.layout + '</i>'); }
 28       if (c.title)      { txt.push('<b>' + c.title + '</b>'); }
 29       if (c.text)       { txt.push('<b>' + c.text + '</b>'); }
 30       if (c.region)     { txt.push('<i>(' + c.region + ')</i>'); }
 31       return (txt.length == 0 ? "Element" : txt.join(" "));
 32   },
 33   
 34   doLoad : function(node,data){
 35     if(data){
 36       node.beginUpdate();
 37       if (!this.designer.isEmptyObject(data)) {
 38         var cs = {
 39              text: this.elementToText(data),
 40              cls: data.items ? 'folder' : 'file' , 
 41              leaf : data.items ? false : true,
 42              jsonId : data[this.jsonId]
 43         };
 44         var cn = node.appendChild(new Ext.tree.TreeNode(cs));
 45         if (data.items) {
 46           for(var i = 0, len = data.items.length; i < len; i++){
 47             this.doLoad(cn,data.items[i]);
 48           }
 49         }
 50       }
 51       node.endUpdate();
 52       return true;
 53     }
 54     return false;
 55   }

237 Ext.ux.grid.PropertyStore = Ext.extend(Ext.grid.PropertyStore, {
238     jsonId : "__JSON__",
239 
240     getPropertyType : function (name) {
241       if (this.grid && this.grid.getPropertyType) return this.grid.getPropertyType(name);
242       return null;
243     },
244 
245 
246     // protected - should only be called by the grid.  Use grid.setSource instead.
247     setSource : function(o){
248         this.source = o;
249         this.store.removeAll();
250         var data = [];
251         for(var k in o){
252             if(k.indexOf(this.jsonId)!=0 && ['items'].indexOf(k)==-1){
253                 if (typeof(o[k]) == 'function') {
254                   data.push(new Ext.grid.PropertyRecord({name: k, value: o[this.jsonId + k] || String(o[k]) , type : 'function'}, k));
255                 } else if (typeof(o[k]) == 'object') {
256                   data.push(new Ext.grid.PropertyRecord({name: k, value: o[this.jsonId + k] || String(Ext.ux.JSON.encode(o[k])), type : 'object'}, k));
257                 } 
258                   data.push(new Ext.grid.PropertyRecord({name: k, value: o[this.jsonId + k] || o[k], type : o[this.jsonId + k] ? 'function' : '' }, k));
259             }
260         }
261         this.store.loadRecords({records: data}, {}, true);
262     },
263     
264     
265     // private
266     onUpdate : function(ds, record, type){
267         if(type == Ext.data.Record.EDIT){
268             var v = record.data['value'];
269             var oldValue = record.modified['value'];
270             if(this.grid.fireEvent('beforepropertychange', this.source, record.id, v, oldValue) !== false){
271                 record.data.changeValue = this.updateSource(record.id,v,record.data.type);
272                 record.commit();
273                 this.grid.fireEvent('propertychange', this.source, record.id, v, oldValue);
274             }else{
275                 record.reject();
276             }
277         }
278     },
279     
280     updateSource : function (prop,value,type) {
281       var propType = this.getPropertyType(prop);
282       if (!type && propType) type=propType.type;
283       if (typeof(this.source[this.jsonId + prop])!='undefined' ||
284        ['object','function','mixed'].indexOf(type)!=-1 ||
285        !propType) {
286          this.source[this.jsonId + prop] = value;
287          try {
288           //Set the jsonScope to be used during eval
289            if (this.grid) var scope = this.grid.jsonScope;
290            var o = eval("( { data :" + value + "})");
291            this.source[prop] = o.data;
292          } catch (e) {Ext.Msg.alert('Exception','Could not set ' + prop + ' to ' + value + '/n' + e);}
293       } else {
294         this.source[prop] = value;
295       }
296       return this.source[prop];
297     },
298 
299     setValue : function(prop, value){
300         this.updateSource(prop,value);
301         this.store.getById(prop).set('value', value);
302     }
303     
304 });

327 Ext.extend(Ext.ux.grid.PropertyColumnModel,Ext.grid.PropertyColumnModel, {
328     // private
329     
330     getPropertyType : function (name) {
331       if (this.grid && this.grid.getPropertyType) return this.grid.getPropertyType(name);
332       return null;
333     },
334     
335     getCellEditor : function(colIndex, rowIndex){
336         var p = this.store.getProperty(rowIndex);
337         var n = p.data['name'], val = p.data['value'], t = p.data['type'];
338         if(this.grid.customEditors[n]){
339             return this.grid.customEditors[n];
340         }
341         var prop = this.getPropertyType(n);
342         if (!t && prop) {
343           t=prop.type;
344           if (!t && prop.values) {
345            var editor =  prop.editable ? this.editors['editlist'] : this.editors['list'];
346            editor.field.setList(prop.values);
347            return editor;
348           }
349         }
350         if (t && this.editors[t]) {
351           return this.editors[t];
352         } else if(Ext.isDate(val)){
353             return this.editors['date'];
354         }else if(typeof val == 'number'){
355             return this.editors['number'];
356         }else if(typeof val == 'boolean'){
357             return this.editors['boolean'];
358         }else{
359             return this.defaultEditor || this.editors[prop ? 'string' : 'mixed'];
360         }
361     },
362     
363     valueRenderer  : function(value, p, r) {
364       if (typeof value == 'boolean') {
365         p.css = (value ? "typeBoolTrue" : "typeBoolFalse");
366         return (value ? "True" : "False");
367       } 
368       var propType = this.getPropertyType(r.id);
369       if (propType && ['object','array','object/array'].indexOf(propType.type)!=-1) {
370         p.css = "typeObject";
371       }
372       return value;
373     },
374     
375     propertyRenderer :  function(value, p) {
376       var propType = this.getPropertyType(value);
377       if (propType) {
378         qtip = propType.desc || '';
379         p.attr = 'qtip="' + qtip.replace(/"/g,'"') + '"'; //'
380       }
381       return value;
382     },
383     
384     getRenderer  : function(col){
385        return col == 0 ? this.propertyRendererDelegate : this.valueRendererDelegate;
386     }
387 });
