 48   * Create node but enabling childeren from Json

 51     var childeren = attr.childeren;
 52     delete attr.childeren;
 53     if(this.baseAttrs){
 54         Ext.applyIf(attr, this.baseAttrs);
 55     }
 56     if(this.applyLoader !== false){
 57         attr.loader = this;
 58     }
 59     if(typeof attr.uiProvider == 'string'){
 60        attr.uiProvider = this.uiProviders[attr.uiProvider] || eval(attr.uiProvider);
 61     }
 62     if (!childeren) {
 63       return(attr.leaf===false ? 
 64             new Ext.tree.AsyncTreeNode(attr) : 
 65             new Ext.tree.TreeNode(attr) );
 66     } else {
 67       var node = new Ext.tree.TreeNode(Ext.applyIf(attr,{draggable:false}));
 68       for(var i = 0, len = childeren.length; i < len; i++){
 69        var n = this.createNode(childeren[i]);
 70        if(n) node.appendChild(n);
 71       }
 72       return node;
 73     }

 99 Ext.extend(Ext.ux.plugin.CookieFiles,Ext.util.Observable,{
100   files : null,
101   last  : null,

104   init : function(){
105     this.cookies = new Ext.state.CookieProvider();     
106     this.files = this.cookies.get('Designer.files') || {};
107   },
108   
109   reset : function(){
110     this.cookies.clear('Designer.files');
111   },
112   
113   saveChanges : function(id,action,callback,content) {
114     this.files[id] = id;
115     this.cookies.set('Designer.files',this.files);
116     if (content) this.cookies.set('Designer/' + id,escape(content));
117     if (action=='delete') {
118       delete this.files[id];
119       this.cookies.clear('Designer.'+id);
120       if (id==this.last) this.last = null;
121     } else {
122       this.last = id;
123     }
124     if(typeof callback == "function") callback(true);
125   },
126 
127   openFile : function(fileInfo,callback) {
128     var result = unescape(this.cookies.get('Designer/' + fileInfo));
129     this.last = fileInfo;
130     if(typeof callback == "function") callback(true,result);
131   },
132   
133   
134   deleteFile : function(fileInfo,callback){
135     this.saveChanges(fileInfo,'delete',callback);
136   },
137   
138   renameFile : function(fileFrom,fileTo,callback){
139     var last = this.last;
140     this.openFile(fileFrom,function(success,content) {
141       if (success) {
142          this.saveChanges(fileTo,'save',function(success){          
143            if (success) {              
144               this.deleteFile(fileFrom,function(success){
145                 if (success && last==fileFrom) this.last=fileTo;
146                 if(typeof callback == "function") callback(success);
147               }.createDelegate(this));
148            } else if(typeof callback == "function") callback(success);
149          }.createDelegate(this),content);
150       } else if(typeof callback == "function") callback(success);
151     }.createDelegate(this));
152   },
153   
154   saveFile : function(fileInfo,content,callback){
155     this.saveChanges(fileInfo,'save',callback,content);
156   },
157   
158   newFile  : function(fileInfo,content,callback){
159     this.saveChanges(fileInfo,'new',callback,content);
160   },
161   
162   load : function(node, callback){ 
163    this.loadNodes(node,false);
164    if(typeof callback == "function")  callback();   
165   },
166   
167   loadNodes : function(node,append){
168     this.activeNode = null;
169     if (!append) while(node.firstChild) node.removeChild(node.firstChild);
170     node.beginUpdate();
171     for (f in this.files){
172         var file = this.files[f];
173         var path = f.split('/');
174         var name = '';
175         var cnode = node;
176         var n;
177         for (var i=0;i<path.length;i++) {
178            name += path[i];
179            n=null;
180            for (var j=0,c=cnode.childNodes;j<c.length && !n;j++) {
181              if (c[j].attributes.text==path[i]) n = c[j];
182            }
183            if (!n) {
184              var leaf = (i==path.length-1);             
185              n = new Ext.tree.TreeNode({
186                      text: (name==this.last ? '<B>' + path[i] + '</B>' : path[i]),
187                      cls:  leaf ? 'file' : 'folder' , 
188                      leaf : leaf,
189                      id  : name
190              }); 
191              cnode.appendChild(n);
192              if (name==this.last) this.activeNode = n;
193            }
194            cnode = n;
195            name += '/' 
196         }
197     }
198     node.endUpdate();
199     return this.activeNode;      
200   }

445   /**
446    * Append the config to the element
447    * @param {Element} el The element to which the config would be added
448    * @param {Object} config The config object to be added
449    * @return {Component} The component added
450    */
451   appendConfig : function (el,config,select,dropLocation,source){
452     if (!el) return false;
453     this.markUndo();
454     
455     //Custom function for adding stuff to a container
456     var add =  function(src,comp,at,before){
457       if(!src.items) src.initItems();
458       var pos = src.items.length;
459       for (var i=0;i<src.items.length;i++) {
460         if (src.items.items[i]==at) {
461           pos = (before) ? i : i+1;
462           i=src.items.length;
463         }
464       }
465       if (!src.codeConfig) src.codeConfig = this.getConfig(src);
466       if (!src.codeConfig.items || !(src.codeConfig.items instanceof Array)) 
467           src.codeConfig.items =  [];
468       delete src.codeConfig.html; //items and html go not together in IE
469       if (pos>src.codeConfig.items.length) 
470         src.codeConfig.items.push(comp)
471       else 
472         src.codeConfig.items.splice(pos, 0, comp);      
473     }.createDelegate(this);
474     
475 
476     if (typeof config == 'function') {
477       config.call(this,function(config) {
478         this.appendConfig(el,config,true);
479       }.createDelegate(this),this);
480     } else {
481      //Get the config of the items
482      var ccmp,cmp= this.getDesignElement(el,true);
483      var items = this.editableJson(this.clone(config));
484      //Find the container that should be changed
485      ccmp = this.getContainer(cmp); 
486      if (dropLocation == 'appendafter') {
487        add(ccmp,items,this.activeElement,false);      
488      } else if (dropLocation == 'appendbefore') { 
489        add(ccmp,items,this.activeElement,true);
490      } else if (dropLocation == 'moveafter') {
491        this.removeElement(source,true);
492        add(ccmp,items,this.activeElement,false);      
493      } else if (dropLocation == 'movebefore') { 
494        this.removeElement(source,true);
495        add(ccmp,items,this.activeElement,true);
496      } else if (dropLocation == 'move') {
497        this.removeElement(source,true);
498        add(ccmp,items);
499      } else // Append default behavior
500        add(ccmp,items);
501      this.modified = true;
502      this.fireEvent('add');
503      this.updateElement(ccmp,null,items[this.jsonId]); 
504     } 
505     return false;

565   getConfig : function (el) {
566     el = el || this.container.items.first();
567     if (!el) return {};

569       var findIn = function(o) {
570         if (!o) return null;
571         if (o[this.jsonId]==el[this.jsonId]) return o;
572         if (o.items) {
573           for (var i=0;i<o.items.length;i++) {
574             var r = findIn(o.items[i]);
575             if (r) return r;
576           }
577         }
578         return null;
579       }.createDelegate(this);
580       el.codeConfig = findIn(this.codeConfig)
581     }

617   /**

623   updateElement : function (element,config,selectId) {
624     var el = element || this.activeElement;
625     if (el) {
626       try {
627         var id = selectId || (this.activeElement ? this.activeElement[this.jsonId] : null);
628         if (config) {Ext.apply(this.getConfig(el),config);}
629         var p = this.container; //Redraw whole canvas          
630         if (!this.redrawContainer && el!=p) {
631            //Check if whe can find parent which can be redraw
632            var c = '';
633            p = this.getContainer(el);
634            //Search if whe find a layout capeble contianer
635            while (p!=this.container && !c) {
636              if (!p.codeConfig) p.codeConfig = this.getConfig(p);
637              c = p.codeConfig.layout;
638              if (!c || (p==el && c)) 
639                 p = this.getContainer(p.ownerCt);
640            }
641            p = c ? p : this.getContainer(el.ownerCt);
642         }
643         this.applyJson(this.getConfig(p).items,p);
644         this.redrawContainer=false;
645         this.selectElement(id);
646       } catch (e) { Ext.Msg.alert('Failure', 'Failed to update element ' + e); }
647       this.fireEvent('change',el);
648       this.modified = true;
649       return true;
650     }
651     return  false;
