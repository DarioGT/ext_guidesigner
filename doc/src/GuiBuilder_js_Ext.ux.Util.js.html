 81      /**
 82       * Convert a url into a nocache url by adding a date if needed
 83       * @param {String} url The url to use
 84       * @param {boolean} nocache The nocache variable (default this.nocache)
 85       */
 86      nocacheUrl : function(url,nocache){
 87        nocache = nocache==undefined ? this.nocache : nocache;
 88        if (nocache) {
 89          url += (url.indexOf('?') != -1 ? '&' : '?') + '_dc=' + (new Date().getTime());
 90        }
 91        return url;
 92      },

 94     /**
 95      * A Synchronized Content loader for data from url
 96      * @param {String} url The url to load synchronized
 97      * @param {Boolean} nocache Should caching of file be disabled
 98      * @param {Boolean} responseXML Should responseXML be returned instead of responseText
 99      * @return {Boolean/String/XMLObject} When there was a error False otherwise response base on responseXML flag
100      */
101     syncContent : function(url,nocache,responseXML) {
102      var activeX = Ext.lib.Ajax.activeX;
103      var isLocal = (document.location.protocol == 'file:');
104      var conn;
105 
106      try {
107       if(Ext.isIE7 && isLocal){throw("IE7forceActiveX");}
108       conn = new XMLHttpRequest();
109      } catch(e)  {
110        for (var i = 0; i < activeX.length; ++i) {
111          try {conn = new ActiveXObject(activeX[i]); break;} catch(e) {}
112        }
113      }
114      //Should we disable caching
115      url = this.nocacheUrl(url,nocache);
116      try {
117       conn.open('GET', url , false);
118       conn.send(null);
119       if ((isLocal && conn.responseText.length!=0) || (conn.status !== undefined && conn.status>=200 && conn.status< 300)) {
120         return responseXML ? conn.responseXML : conn.responseText;
121       }
122      } catch (e) {
123        this.fireEvent('error','synchronized',e);
124      }
125      return false;
126     },

128     /**
129      * Function used to load a JavaScript into a document.head element
130      * the id of the script item is the name of the file.
131      * @param {String} url The url to load javascript from
132      * @param {Boolean} nocache Should caching of javascript files be disabled
133      * @return {Boolean} Indicator if load went corretly true/false
134      */
135     scriptLoader : function(url,nocache) {
136      if (!url) return false;
137      var id=url;
138      if(!document.getElementById(id)) {
139        var content = this.syncContent(url,nocache);
140        if (content===false) return false;
141        var head = document.getElementsByTagName("head")[0];
142        var script = document.createElement("script");
143        try {
144          script.text = content;
145        } catch (e) {
146          script.appendChild(content);
147        }
148        script.setAttribute("type", "text/javascript");
149        script.setAttribute("id", id);
150        head.appendChild(script);
151      }
152      return true;
153     },

155     /**
156      * Get the value of a url action
157      * @param {String} name The name of the action
158      * @param {String} defaultValue The default value used when not found
159      * @param {String} url The url to use defaults to window url
160      * @return {String} The value found for action or default
161      */
162     getUrlAction : function (name,defaultValue,url) {
163       name = name.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
164       var regexS = "[\\?&]"+name+"=([^&#]*)";
165       var regex = new RegExp( regexS );
166       var results = regex.exec( url || window.location.href );
167       if( results == null )
168         return isNaN(defaultValue) ? defaultValue : Number(defaultValue);
169       else
170         return isNaN(results[1]) ? results[1] : Number(results[1]);
171     },

173     /**
174      * Parse a url into object
175      * containing element: "source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","name","ext","query","anchor"
176      * @param {String} str The url to parse
177      * @param {Object} options A object with options to be used during parse
178      */
179     parseUrl : function(str,options) {
180       var o   = Ext.applyIf(options || {},{
181           strictMode: false,
182           key: ["source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","query","anchor"],
183           q:   {
184             name:   "queryKey",
185             parser: /(?:^|&)([^&=]*)=?([^&]*)/g
186           },
187           parser: {
188             strict: /^(?:([^:\/?#]+):)?(?:\/\/((?:(([^:@]*):?([^:@]*))?@)?([^:\/?#]*)(?::(\d*))?))?((((?:[^?#\/]*\/)*)([^?#]*))(?:\?([^#]*))?(?:#(.*))?)/,
189             loose:  /^(?:(?![^:@]+:[^:@\/]*@)([^:\/?#.]+):)?(?:\/\/)?((?:(([^:@]*):?([^:@]*))?@)?([^:\/?#]*)(?::(\d*))?)(((\/(?:[^?#](?![^?#\/]*\.[^?#\/.]+(?:[?#]|$)))*\/?)?([^?#\/]*))(?:\?([^#]*))?(?:#(.*))?)/
190           }
191         }),
192         m   = o.parser[o.strictMode ? "strict" : "loose"].exec(str),
193         uri = {},
194         i   = 14;
195 
196       while (i--) uri[o.key[i]] = m[i] || "";
197       uri[o.q.name] = {};
198       uri[o.key[12]].replace(o.q.parser, function ($0, $1, $2) {
199         if ($1) uri[o.q.name][$1] = $2;
200       });
201       if (!uri.protocol) {
202         uri.relative= uri.host + uri.relative;
203         uri.path = uri.host + uri.path;
204         uri.directory = uri.host + uri.directory;
205         uri.host="";
206       }
207       if (!uri.file) {
208         uri.file = uri.path;
209         uri.path = uri.directory = '';
210       }
211       uri['name']="";uri['ext']="";
212         if (/\.\w+$/.test(uri.file) && uri.file.match(/([^\/\\]+)\.(\w+)$/)) {
213           uri['name']=RegExp.$1;
214           uri['ext']=RegExp.$2;
215         }
216       return uri;
217     },

219     /**
220      * Load a url, stylesheet or javascript by adding it to the header
221      * @param {String} url The url to be added
222      * @param {String} type The load action to perform (default js)
223      * @param {Object} options options[reload] Should the file be reload if allready loaded
224      */
225     loadUrl : function(url,type,options) {
226       options = options || {};
227       //Now load it by adding it to the header
228       if (this._loadCount == undefined) {
229         this._loadCount=0;
230         this._loaded=[];
231       }
232       if (this._loaded[url] && !options.reload) return;
233       var node;
234       var self = this;
235       var head = document.getElementsByTagName("head")[0];
236       if(document.getElementById(url)){head.removeChild(url)}
237       switch (type) {
238        case 'css' :
239          node = document.createElement("link");
240          node.setAttribute("rel", "stylesheet");
241          node.setAttribute("type", "text/css");
242          node.setAttribute("href", this.nocacheUrl(url,options.nocache));
243          break;
244        default :
245          node = document.createElement("script");
246          node.setAttribute("type", "text/javascript");
247          node.setAttribute("src", this.nocacheUrl(url,options.nocache));
248       }
249       node.setAttribute("id",url);
250       node.onload = node.onreadystatechange = function() {
251        if (!self._loaded[url] && (!node.readyState || [4,"loaded","complete"].indexOf(node.readyState)!=-1)) {
252          self._loadCount--;
253          self._loaded[url]=true;
254        }
255       };
256       this._loadCount++;
257       this._loaded[url]=false;
258       head.appendChild(node);
259     },

261     /**
262      * Function require is used to checks if the required files are loaded
263      * when not the files are loaded
264      * @param {Mixed} packages An array or ; seperated string with packages to load
265      * @param {Mixed} options An object with options to used, when string then basedir is assumed
266      * options[basedir] The default directory to use for all files
267      * options[cssdir] The directory to be used for stylesheets
268      * options[async] When set to true required will return directly
269      * options[callback] Callback function after all required files are loaded
270      * options[reload] When reload is set packages are reloaded
271      * options[nocache] When set the object caching is turned off
272      * options[reload] When true stylesheet/javacode is swapped
273      * @return {Object} a object with keys js and css contain an array with full path of items
274      */
275     require : function(packages,options){
276       var ret = {css: [],js:[]};
277       if (!packages) return ret;
278       if (packages instanceof Array && !options && packages.length==2 && typeof(packages[1])=='object') {
279         options=packages[1];
280         packages=packages[0];
281       }
282       packages= (typeof(packages)=='string') ? packages.split(';') : packages || [];
283       options = (typeof(options)=='string') ? {'basedir' : options} : options || {basedir : ""};
284       options['cssdir'] = options.cssdir || options.basedir;
285       options['reload'] = options['reload']== undefined ? true : options['reload'];
286       options['nocache'] = options['nocache']== undefined ? this.nocache : options['nocache'];
287       //First load all files
288       for (var i=0;i<packages.length;i++) {
289         //Create a name of path + packages + extentsion
290         var url = this.parseUrl(packages[i]);
291         if (['js','css'].indexOf(url.ext)==-1) url.file+='.js';
292         var dir = url.ext=='css' ? options.cssdir : options.basedir;
293         var uri = dir + (dir.length!=0 && dir.charAt(dir.length-1)!='/' ? "/" : "") +
294            url.directory +  url.file + (url.query ? '?' + url.query : '');
295         if (url.ext=='css') {
296             ret.css.push(uri);
297             Ext.util.CSS.swapStyleSheet(this.nocacheUrl(uri,options.nocache), uri);
298         } else if (options.async){
299            this.loadUrl(uri,url.ext,options);
300            ret.js.push(uri);
301         } else {
302            ret.js.push(uri);
303            this.scriptLoader(uri,options.nocache);
304         }
305       }
306       return ret;
307     }
