 20 //Register name spaces used
 21 Ext.namespace('Ext.ux.guid');
 22 Ext.namespace('Ext.ux.guid.data');  
 23 
 24 /*
 25  * CookieFiles
 26  */
 27 Ext.ux.guid.data.CookieRepository = Ext.extend(Ext.ux.guid.data.Repository,{
 28   
 29   init : function(){
 30     this.cookies = new Ext.state.CookieProvider();     
 31     Ext.ux.guid.data.CookieRepository.superclass.init.call(this);
 32   },
 33   
 34   refresh : function (callback) {
 35     this.items = this.cookies.get('repository.files');
 36     Ext.ux.guid.data.CookieRepository.superclass.refresh.call(this,callback);
 37   },
 38 
 39   saveChanges : function(id,action,callback,content) {  
 40     if (content) this.cookies.set('repository/' + id,escape(content));
 41     if (action=='remove') this.cookies.clear('repository/'+id);
 42     Ext.ux.guid.data.CookieRepository.superclass.saveChanges.call(this,id,action,callback,content);
 43     this.cookies.set('repository.files',this.items);
 44   },
 45 
 46   open : function(id,callback) {
 47     var content = unescape(this.cookies.get('repository/' + id));
 48     Ext.ux.guid.data.CookieRepository.superclass.open.call(this,id,callback,content)
 49   }
 50     
 51 });
